-- Create EMP45 table
CREATE TABLE EMP45 (
    EMPID     NUMBER(6) PRIMARY KEY,
    LNAME     VARCHAR2(50) NOT NULL,
    FNAME     VARCHAR2(50) NOT NULL,
    SALARY    NUMBER(8, 2) CONSTRAINT SALARY_CK CHECK (SALARY <= 20000),
    COMMPCT   NUMBER(4, 2),
    HIREDATE  DATE
);

-- PL/SQL block to process TRANS45 table
DECLARE
    v_salary EMP45.SALARY%TYPE;
    v_action CHAR(1);
    v_empid  TRANS45.EMPID%TYPE;
    v_error  VARCHAR2(100);
BEGIN
    FOR rec IN (SELECT * FROM TRANS45) LOOP
        BEGIN
            -- Process action
            v_action := rec.ACTION;
            v_empid := rec.EMPID;

            IF v_action = 'U' THEN
                -- Update salary
                UPDATE EMP45
                SET SALARY = rec.SALARY
                WHERE EMPID = v_empid;
                IF SQL%ROWCOUNT = 0 THEN
                    RAISE_APPLICATION_ERROR(-20001, 'Employee does not exist.');
                END IF;

            ELSIF v_action = 'I' THEN
                -- Increase salary
                UPDATE EMP45
                SET SALARY = SALARY + rec.SALARY
                WHERE EMPID = v_empid;
                IF SQL%ROWCOUNT = 0 THEN
                    RAISE_APPLICATION_ERROR(-20002, 'Employee does not exist.');
                END IF;

            ELSIF v_action = 'D' THEN
                -- Decrease salary
                UPDATE EMP45
                SET SALARY = SALARY - rec.SALARY
                WHERE EMPID = v_empid;
                IF SQL%ROWCOUNT = 0 THEN
                    RAISE_APPLICATION_ERROR(-20003, 'Employee does not exist.');
                END IF;

            ELSE
                -- Invalid action
                RAISE_APPLICATION_ERROR(-20004, 'Invalid action specified.');
            END IF;

        EXCEPTION
            WHEN OTHERS THEN
                v_error := SQLERRM;
                DBMS_OUTPUT.PUT_LINE('Error for EMPID ' || v_empid || ': ' || v_error);
        END;
    END LOOP;
END;
/
